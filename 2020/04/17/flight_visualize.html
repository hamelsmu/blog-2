import altair as alt
test = df.groupby(['date'])['callsign'].nunique().reset_index()
alt.Chart(test).mark_line().encode(
    x=alt.X(field='date',type='nominal'),
    y=alt.Y(field='callsign', title='number of flights',type='quantitative'),
    # color=alt.Color(field ='date',type='ordinal'),
